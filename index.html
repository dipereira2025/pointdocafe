async function enviarPedidoFinal() { const n = document.getElementById('cli-nome').value; const e = document.getElementById('cli-end').value; if(!n || !e) return alert("Por favor, preencha Nome e Endereço!"); const selecionados = produtosLocal.filter(p => p.qty > 0); const total = selecionados.reduce((acc, p) => acc + ((p.precoPromo || p.preco) * p.qty), 0); // Objeto do Pedido para o Relatório const pedidoDados = { cliente: n, endereco: e, referencia: document.getElementById('cli-ref').value, pagamento: document.getElementById('cli-pgto').value, observacao: document.getElementById('cli-obs').value, itens: selecionados.map(p => ({ nome: p.nome, qty: p.qty, preco: (p.precoPromo || p.preco) })), total: total, data: new Date().toISOString(), // Data e hora exata dataLocal: new Date().toLocaleDateString('pt-BR'), // Data simplificada para filtro status: "Novo" }; try { // SALVA NO FIREBASE (Pasta 'vendas') await fetch(DB_BASE + "vendas/" + Date.now() + ".json", { method: 'PUT', body: JSON.stringify(pedidoDados) }); // GERA MENSAGEM WHATSAPP let msg = `*☕ NOVO PEDIDO - POINT DO CAFÉ*\n\n`; msg += `*Cliente:* ${n}\n*Pagamento:* ${pedidoDados.pagamento}\n\n`; msg += `*ITENS:*\n`; selecionados.forEach(p => { msg += `• ${p.qty}x ${p.nome} = R$ ${((p.precoPromo || p.preco) * p.qty).toFixed(2)}\n`; }); msg += `\n*TOTAL: R$ ${total.toFixed(2)}*`; window.open(`https://wa.me/${WHATSAPP}?text=${encodeURIComponent(msg)}`); } catch (err) { alert("Erro ao registrar pedido. Tente novamente."); } }
